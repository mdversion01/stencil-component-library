# Autocomplete Multiselect

<Description of={AutocompleteMultiselectStories} />

## HTML

```html
<autocomplete-multiselect
  input-id="acm-example"
  label="Favorite fruits"
  placeholder="Type to search"
  editable
  add-btn
  badge-variant="primary"
></autocomplete-multiselect>

<script>
  const el = document.querySelector('autocomplete-multiselect')

  el.options = [
    'Apple',
    'Banana',
    'Orange',
    'Mango',
    'Blueberry',
    'Strawberry'
  ]

  el.value = ['Apple', 'Mango']

  el.addEventListener('multiSelectChange', (event) => {
    console.log('multiSelectChange:', event.detail)
  })

  el.addEventListener('optionsChange', (event) => {
    console.log('optionsChange:', event.detail)
  })
</script>
```
