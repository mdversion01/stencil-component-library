## HTML

```html
<dropdown-component id="dropdown-basic" button-text="Dropdown" variant="primary"></dropdown-component>

<script>
  function baseItems() {
    return [
      { name: 'Action', value: 'action' },
      { name: 'Another action', value: 'another' },
      { name: 'Something else here', value: 'else' },
    ];
  }

  const el = document.getElementById('dropdown-basic');
  if (el) {
    el.options = baseItems();
  }
</script>
```
