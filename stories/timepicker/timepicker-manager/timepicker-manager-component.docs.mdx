## HTML

```html

<script>
  const el = document.querySelector('timepicker-manager');

  el.addEventListener('managerTimeChange', event => {
    console.log('managerTimeChange', event.detail);
  });

  el.addEventListener('managerTimeInput', event => {
    console.log('managerTimeInput', event.detail);
  });
</script>
```
