# Toasts Component

<Description of={ToastStories} />

## HTML

```html
<section>
  <button type="button" id="show-toast-btn">Show Toast</button>

  <toasts-component id="toast-example" aria-label="Notifications" position="top-right" append-toast persistent></toasts-component>
</section>

<script>
  (async () => {
    await customElements.whenDefined('toasts-component');

    const host = document.getElementById('toast-example');
    const btn = document.getElementById('show-toast-btn');

    btn.addEventListener('click', async () => {
      await host.showToast({
        toastTitle: 'Saved',
        content: 'Your changes were saved successfully.',
        variantClass: 'success',
        svgIcon: 'check-circle-fill',
        additionalHdrContent: 'Just now',
        persistent: true,
      });
    });
  })();
</script>
```
