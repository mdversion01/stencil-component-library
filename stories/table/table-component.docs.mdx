# Table Component

<Description of={TableStories} />

## HTML

```html
<table-component
  id="table-example"
  table-id="table-example"
  responsive
  striped
  sortable
  pagination-enabled
  pagination-variant="standard"
  pagination-position="bottom"
  rows-per-page="10"
  size="sm"
></table-component>

<script>
  (async () => {
    await customElements.whenDefined('table-component');
    const table = document.getElementById('table-example');

    table.items = [
      { last_name: 'MacDonald', first_name: 'Thor', age: 40 },
      { last_name: 'Smith', first_name: 'Anna', age: 25 },
      { last_name: 'Peterson', first_name: 'Zachary', age: 36 },
    ];

    table.originalItems = [...table.items];

    table.fields = [
      { key: 'last_name', label: 'Last Name', sortable: true },
      { key: 'first_name', label: 'First Name', sortable: true },
      { key: 'age', label: 'Age', sortable: true },
    ];
  })();
</script>
```
