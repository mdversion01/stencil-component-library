<Description of={PlumageAutocompleteMultipleSelectionsStories} />

## HTML

```html
<plumage-autocomplete-multiple-selections-component
  input-id="acms-example"
  label="Favorite fruits"
  placeholder="Type to search"
  editable
  add-btn
  badge-variant="primary"
></plumage-autocomplete-multiple-selections-component>

<script>
  const el = document.querySelector('plumage-autocomplete-multiple-selections-component')

  el.options = [
    'Apple',
    'Banana',
    'Orange',
    'Mango',
    'Blueberry',
    'Strawberry'
  ]

  el.value = ['Apple', 'Mango']

  el.addEventListener('multiSelectChange', (event) => {
    console.log('multiSelectChange:', event.detail)
  })
</script>
```
