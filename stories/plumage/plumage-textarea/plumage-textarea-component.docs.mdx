## HTML

```html
<plumage-textarea-component
  label="Message"
  input-id="message-example"
  placeholder="Enter your message"
  rows="4"
></plumage-textarea-component>

<plumage-textarea-component
  label="Description"
  input-id="description-example"
  rows="5"
  max-length="250"
  value="Initial textarea value"
></plumage-textarea-component>

<plumage-textarea-component
  label="Read Only Notes"
  input-id="readonly-notes-example"
  read-only
  value="This field is read only."
  rows="4"
></plumage-textarea-component>

<script>
  const examples = document.querySelectorAll('plumage-textarea-component');

  examples.forEach(el => {
    el.addEventListener('valueChange', event => {
      console.log('valueChange:', event.detail);
    });

    el.addEventListener('blurChange', event => {
      console.log('blurChange:', event.detail);
    });
  });
</script>
```
