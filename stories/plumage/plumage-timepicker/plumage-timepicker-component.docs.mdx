<Description of={PlumageTimepickerStories} />

## HTML

```html
<plumage-timepicker-component
  show-label
  label-text="Meeting time"
  input-id="time-example"
  input-name="meetingTime"
  value="13:05:09"
  is-twenty-four-hour-format="true"
></plumage-timepicker-component>

<script>
  const el = document.querySelector('plumage-timepicker-component');

  el.addEventListener('timeChange', (event) => {
    console.log('timeChange:', event.detail);
  });

  el.addEventListener('timeInput', (event) => {
    console.log('timeInput:', event.detail);
  });
</script>
```
