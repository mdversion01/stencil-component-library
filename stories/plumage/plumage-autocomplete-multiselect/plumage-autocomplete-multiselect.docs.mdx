# Plumage Autocomplete Multiselect

<Description of={PlumageAutocompleteMultiselectStories} />

## HTML

```html
<plumage-autocomplete-multiselect-component
  input-id="ams-example"
  label="Favorite fruits"
  placeholder="Type to search"
  editable
  add-btn
  badge-variant="primary"
></plumage-autocomplete-multiselect-component>

<script>
  const el = document.querySelector('plumage-autocomplete-multiselect-component')

  el.options = [
    'Apple',
    'Banana',
    'Orange',
    'Mango',
    'Blueberry',
    'Strawberry'
  ]

  el.value = ['Apple', 'Mango']

  el.addEventListener('multiSelectChange', (event) => {
    console.log('multiSelectChange:', event.detail)
  })
</script>
```
